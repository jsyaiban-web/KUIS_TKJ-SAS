let questions = [];
let currentPage = 0;
const questionsPerPage = 20; // 20 soal per halaman
const totalPages = 3;        // Halaman 1 (20), Halaman 2 (20), Halaman 3 (10) = Total 50 Soal
let userScores = {}; 

function shuffle(array) {
    for (let i = array.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [array[i], array[j]] = [array[j], array[i]];
    }
    return array;
}

function initQuiz() {
    questions = shuffle([...rawQuestions]);
    currentPage = 0;
    userScores = {}; 
    renderPages();
    showPage(0);
    document.getElementById('quizForm').style.display = 'block';
    document.getElementById('progressArea').style.display = 'block';
    document.getElementById('resultArea').style.display = 'none';
}

function checkInstantAnswer(radioInput, globalIndex, chosenOpt, correctOpt) {
    const block = document.getElementById(`block-${globalIndex}`);
    const feedback = document.getElementById(`fb-${globalIndex}`);
    const optionsGrid = block.querySelector('.options-grid');
    
    optionsGrid.classList.add('disabled-options');
    
    const labels = optionsGrid.querySelectorAll('.option-label');
    
    if (chosenOpt === correctOpt) {
        feedback.innerText = "✅ Benar";
        feedback.className = "feedback-message feedback-correct";
        radioInput.parentElement.classList.add('correct-choice');
        userScores[globalIndex] = 1; 
    } else {
        feedback.innerText = `❌ Salah (Jawaban Benar: ${correctOpt})`;
        feedback.className = "feedback-message feedback-wrong";
        radioInput.parentElement.classList.add('wrong-choice');
        
        labels.forEach(label => {
            if (label.innerText.trim().startsWith(correctOpt)) {
                label.classList.add('correct-choice');
            }
        });
        userScores[globalIndex] = 0; 
    }
}

function renderPages() {
    const pagesContainer = document.getElementById('quizPages');
    pagesContainer.innerHTML = '';

    for (let p = 0; p < totalPages; p++) {
        const pageDiv = document.createElement('div');
        pageDiv.className = `page page-${p}`;
        
        const startIdx = p * questionsPerPage;
        const endIdx = startIdx + questionsPerPage;
        const pageQuestions = questions.slice(startIdx, endIdx);

        pageQuestions.forEach((q, index) => {
            const globalIndex = startIdx + index;
            const qBlock = document.createElement('div');
            qBlock.className = 'question-block';
            qBlock.id = `block-${globalIndex}`;
            qBlock.innerHTML = `
                <div class="question-text">${globalIndex + 1}. ${q.q}</div>
                <div class="options-grid">
                    <label class="option-label"><input type="radio" name="q${globalIndex}" value="A" onclick="checkInstantAnswer(this, ${globalIndex}, 'A', '${q.ans}')"> A. ${q.a}</label>
                    <label class="option-label"><input type="radio" name="q${globalIndex}" value="B" onclick="checkInstantAnswer(this, ${globalIndex}, 'B', '${q.ans}')"> B. ${q.b}</label>
                    <label class="option-label"><input type="radio" name="q${globalIndex}" value="C" onclick="checkInstantAnswer(this, ${globalIndex}, 'C', '${q.ans}')"> C. ${q.c}</label>
                    <label class="option-label"><input type="radio" name="q${globalIndex}" value="D" onclick="checkInstantAnswer(this, ${globalIndex}, 'D', '${q.ans}')"> D. ${q.d}</label>
                </div>
                <div class="feedback-message" id="fb-${globalIndex}"></div>
            `;
            pageDiv.appendChild(qBlock);
        });

        pagesContainer.appendChild(pageDiv);
    }
}

function showPage(pageIndex) {
    const pages = document.querySelectorAll('.page');
    pages.forEach((p, idx) => {
        p.classList.toggle('active', idx === pageIndex);
    });

    currentPage = pageIndex;
    
    document.getElementById('btnPrev').style.display = pageIndex === 0 ? 'none' : 'block';
    if (pageIndex === totalPages - 1) {
        document.getElementById('btnNext').innerText = 'Kirim Jawaban';
    } else {
        document.getElementById('btnNext').innerText = 'Berikutnya';
    }

    const progressPercent = Math.round((pageIndex / totalPages) * 100);
    document.getElementById('progressBar').style.width = `${progressPercent}%`;
    document.getElementById('progressText').innerText = `Halaman ${pageIndex + 1} dari ${totalPages}`;
    document.getElementById('percentText').innerText = `${progressPercent}% Selesai`;
    
    window.scrollTo({top: 0, behavior: 'smooth'});
}

function changePage(direction) {
    if (direction === 1 && currentPage === totalPages - 1) {
        calculateScore();
    } else {
        const nextPage = currentPage + direction;
        if (nextPage >= 0 && nextPage < totalPages) {
            showPage(nextPage);
        }
    }
}

function calculateScore() {
    let totalCorrect = 0;
    let answeredQuestions = 0;
    
    for (let i = 0; i < questions.length; i++) {
        if (userScores[i] !== undefined) {
            answeredQuestions++;
            if (userScores[i] === 1) {
                totalCorrect++;
            }
        }
    }

    if(answeredQuestions < questions.length) {
        const konfirmasi = confirm("Ada soal yang belum Anda klik/jawab. Apakah Anda yakin ingin mengirim dan mengakhiri ujian sekarang?");
        if(!konfirmasi) return;
    }

    const finalScore = Math.round((totalCorrect / questions.length) * 100);
    
    document.getElementById('quizForm').style.display = 'none';
    document.getElementById('progressArea').style.display = 'none';
    
    const resultArea = document.getElementById('resultArea');
    resultArea.style.display = 'block';
    
    document.getElementById('finalScore').innerText = finalScore;
    
    let msg = "";
    if (finalScore >= 85) msg = "Kompeten! Anda memiliki pemahaman sistem operasi dan server TKJ yang luar biasa.";
    else if (finalScore >= 75) msg = "Bagus! Anda lulus standar kompetensi dasar sysadmin TKJ.";
    else msg = "Nilai Anda di bawah KKM. Silakan pelajari kembali log koreksi instan di atas dan coba lagi.";
    
    document.getElementById('scoreMessage').innerText = msg;
    window.scrollTo({top: 0, behavior: 'smooth'});
}
